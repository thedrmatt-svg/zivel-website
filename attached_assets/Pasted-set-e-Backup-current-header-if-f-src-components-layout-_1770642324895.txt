"use client";

import Link from "next/link";
import { usePathname } from "next/navigation";

import { services } from "@/lib/data/services";
import { pathways } from "@/lib/data/pathways";

function NavLink({ href, label }: { href: string; label: string }) {
  const pathname = usePathname();
  const active = pathname === href || (href !== "/" && pathname.startsWith(href));
  return (
    <Link
      href={href}
      className={[
        "text-sm font-medium transition",
        active ? "text-white" : "text-white/70 hover:text-white",
      ].join(" ")}
    >
      {label}
    </Link>
  );
}

function Dropdown({
  label,
  items,
  viewAllHref,
  viewAllLabel,
}: {
  label: string;
  items: { href: string; title: string; description?: string }[];
  viewAllHref: string;
  viewAllLabel: string;
}) {
  return (
    <div className="relative group">
      <button
        type="button"
        className="text-sm font-medium text-white/70 hover:text-white transition inline-flex items-center gap-2"
        aria-haspopup="menu"
        aria-expanded="false"
      >
        {label}
        <span className="text-white/40 group-hover:text-white/70">v</span>
      </button>

      <div className="pointer-events-none opacity-0 translate-y-1 group-hover:pointer-events-auto group-hover:opacity-100 group-hover:translate-y-0 transition absolute left-0 top-full mt-3 w-[420px] z-50">
        <div className="rounded-2xl border border-white/10 bg-black/90 backdrop-blur p-3 shadow-xl">
          <div className="grid gap-2">
            {items.map((it) => (
              <Link
                key={it.href}
                href={it.href}
                className="rounded-xl border border-white/10 bg-white/5 hover:bg-white/10 hover:border-white/20 p-4 transition"
              >
                <div className="text-sm font-semibold text-white">{it.title}</div>
                {it.description ? (
                  <div className="mt-1 text-xs text-white/65">{it.description}</div>
                ) : null}
              </Link>
            ))}
          </div>

          <div className="mt-2 pt-2 border-t border-white/10">
            <Link
              href={viewAllHref}
              className="inline-flex text-sm font-medium text-[var(--zivel-gold)] hover:opacity-90"
            >
              {viewAllLabel} ->
            </Link>
          </div>
        </div>
      </div>
    </div>
  );
}

          <img

            alt="Zivel"
            className="h-7 w-auto"
          />
        </Link>

        {/* Desktop nav */}
        <nav className="hidden lg:flex items-center gap-6">
          <Dropdown

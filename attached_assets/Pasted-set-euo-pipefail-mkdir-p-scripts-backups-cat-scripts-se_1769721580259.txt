set -euo pipefail
mkdir -p scripts .backups

cat > scripts/set-cryo-vars-in-css.mjs <<'EOF'
import fs from "node:fs";
import path from "node:path";

const FILE = "src/styles/globals.css";
const START = "/* ZIVEL_SERVICE_THEME_START */";
const END   = "/* ZIVEL_SERVICE_THEME_END */";

function backup(filePath){
  const stamp = new Date().toISOString().replace(/[:.]/g,"-");
  const safe = filePath.replace(/\//g,"__");
  const out = path.join(".backups", `${safe}.${stamp}.bak`);
  fs.copyFileSync(filePath, out);
  return out;
}

function main(){
  if (!fs.existsSync(FILE)) {
    console.error("❌ Not found:", FILE);
    process.exit(1);
  }

  const b = backup(FILE);
  let css = fs.readFileSync(FILE, "utf8");

  const blockRe = new RegExp(`${START}[\\s\\S]*?${END}`, "m");
  const m = css.match(blockRe);
  if (!m) {
    console.error("❌ Theme block markers not found in src/styles/globals.css");
    process.exit(1);
  }

  let block = m[0];

  // 1) Ensure defaults exist (so other pages remain neutral)
  if (!block.includes("--zivel-service-r")) {
    // add safe defaults inside .zivel-service-page
    block = block.replace(
      /\.zivel-service-page\s*\{\s*/m,
      `.zivel-service-page{\n  --zivel-service-r: 0;\n  --zivel-service-g: 0;\n  --zivel-service-b: 0;\n  `
    );
  }

  // 2) Add Cryotherapy mapping based on the data attribute
  const cryoRule = `.zivel-service-page[data-zivel-service="cryotherapy"]{\n  --zivel-service-r: 45;\n  --zivel-service-g: 100;\n  --zivel-service-b: 189;\n}\n`;

  if (!block.includes(`data-zivel-service="cryotherapy"`)) {
    // insert right after the .zivel-service-page rule (after its closing brace)
    // find the first occurrence of ".zivel-service-page{...}\n}"
    const idx = block.indexOf(".zivel-service-page");
    if (idx === -1) {
      console.error("❌ Could not find .zivel-service-page rule inside theme block");
      process.exit(1);
    }

    // naive but reliable: insert after the first closing brace following .zivel-service-page{
    const after = block.indexOf("}", idx);
    if (after === -1) {
      console.error("❌ Could not parse .zivel-service-page rule");
      process.exit(1);
    }

    block = block.slice(0, after + 1) + "\n\n" + cryoRule + block.slice(after + 1);
  }

  // 3) Replace the theme block in the file
  css = css.replace(blockRe, block);
  fs.writeFileSync(FILE, css, "utf8");

  console.log("✅ Added Cryotherapy RGB vars via CSS attribute selector");
  console.log("File:", FILE);
  console.log("Backup:", b);
}

main();
EOF

node scripts/set-cryo-vars-in-css.mjs

echo ""
echo "=== VERIFY cryo vars exist ==="
rg -n "data-zivel-service=\"cryotherapy\"|--zivel-service-r" src/styles/globals.css

echo ""
echo "=== BUILD ==="
npm run build
